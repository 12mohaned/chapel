\lstdefinelanguage{syntax}
  {
    morekeywords={one, of},
    sensitive=false,
    mathescape=true,
    basicstyle=\footnotesize\tt,
    keywordstyle=\footnotesize\tt,
    commentstyle=\footnotesize\tt,
    identifierstyle=\footnotesize\it,
    morecomment=[s]{`}{'},
    literate={[[}{{$[$}}{1}
             {]]}{{$]$}}{1}
             {[[,...]]}{{$[$, $\ldots]$}}{5}
             {[[...]]}{{$[\ldots]$}}{3}
  }
